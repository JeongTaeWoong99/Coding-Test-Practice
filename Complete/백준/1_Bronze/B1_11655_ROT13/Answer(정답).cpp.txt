#include <bits/stdc++.h>
using namespace std;

// 전역 변수
string S;    // 입력 문자열 (공백 포함)
string temp; // 암호화된 결과 문자열

int main()
{
    ios_base::sync_with_stdio(false);cin.tie(0);cout.tie(0);

    // 공백 포함 입력
    getline(cin, S);

    // 각 문자를 ROT13 암호화
    for(int i = 0; i < S.length(); i++)
    {
        if      (S[i] >= 'A' && S[i] <= 'Z') temp += ((S[i] - 'A' + 13) % 26 + 'A'); // 대문자 ROT13
        else if (S[i] >= 'a' && S[i] <= 'z') temp += ((S[i] - 'a' + 13) % 26 + 'a'); // 소문자 ROT13
        else                                 temp += S[i];                           // 공백 및 기타 문자 그대로
    }

    // 결과 출력
    cout << temp;

    return 0;
}
