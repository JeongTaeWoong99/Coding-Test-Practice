#include <bits/stdc++.h>
using namespace std;

int         temp;
vector<int> v;

int main()
{
    // 4개의 정수 입력
    for (int i = 0; i < 4; i++)
    {
        cin >> temp;
        v.emplace_back(temp);
    }

    // 정렬
    sort(v.begin(), v.end());

    // 첫 번째와 세 번째 값을 곱하면 최대 직사각형
    cout << v[0] * v[2];

    return 0;
}
